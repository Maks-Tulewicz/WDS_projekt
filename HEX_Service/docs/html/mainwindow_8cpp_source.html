163void MainWindow::updateServoGUI(const ServoFrame &frame)
164{
165 // 1) Update QLabel table
166 for (int i = 0; i < frame.angles.size(); ++i) {
167 auto name = QString("servo_%1_%2").arg(i/3).arg(i%3);
168 if (auto *lbl = findChild<QLabel*>(name))
169 lbl->setText(QString::number(frame.angles[i], 'f', 1) + "°");
170 }
171 // 2) TopView
172 for (int leg = 0; leg < 6; ++leg) {
173 float hip = frame.angles[leg*3 + 0];
174 topView->setHipAngle(leg, hip - 90.0f);
175 }
176 // 3) SideView
177 int legIdx = ui->comboBoxLegSide->currentIndex();
178 float knee = frame.angles[legIdx*3 + 1];
179 float ankle = frame.angles[legIdx*3 + 2];
180 sideView->setJointAngles(knee, ankle);
181
182 QString frameText;
183 frameText += QString("Time: %1 ms\n").arg(frame.timeMs);
184 frameText += "Angles: [";
185 for (int i = 0; i < frame.angles.size(); ++i) {
186 frameText += QString::number(frame.angles[i], 'f', 1);
187 if (i < frame.angles.size() - 1)
188 frameText += ", ";
189 }
190 frameText += "]\n";
191 frameText += QString("Speed: %1 pts/s\n").arg(frame.speed, 0, 'f', 1);
192 frameText += QString("Packets: %1\n").arg(frame.packetCount);
193 frameText += "---------------------------";
194
195 logToTerminal(frameText);
196}
